# List aliases:
$ alias
diff='diff --color=auto'

# Create temporary alias:
$ alias grep="grep --color=auto -P"

# Create permanent alias:
$ code ~/.zshrc

# Display alias:
$ alias ls
ls='ls --color=auto'

# Escape alias (technique 1), use 'command':
$ command ls

# Escape alias (technique 2), escape a letter:
$ \ls
